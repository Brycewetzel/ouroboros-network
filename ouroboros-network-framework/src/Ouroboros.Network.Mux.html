               ( Show failure

               , ShowProxy ps
               )

            -> Codec ps failure m bytes
            -> Peer ps pr st m a

            -> Codec ps failure m bytes
            -> PeerPipelined ps pr st m a

    MuxPeerRaw
           :: (Channel m bytes -> m (a, Maybe bytes))
           -> MuxPeer bytes m a
