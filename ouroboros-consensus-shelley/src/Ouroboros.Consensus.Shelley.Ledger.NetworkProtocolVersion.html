instance HasNetworkProtocolVersion (ShelleyBlock era) where
  type BlockNodeToNodeVersion   (ShelleyBlock era) = ShelleyNodeToNodeVersion
  type BlockNodeToClientVersion (ShelleyBlock era) = ShelleyNodeToClientVersion
