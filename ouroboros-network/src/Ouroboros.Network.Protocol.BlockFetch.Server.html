    :: m (BlockFetchSendBlocks block m a)

    :: m (BlockFetchServer block m a)
    -> BlockFetchBlockSender block m a

data BlockFetchSendBlocks block m a where

    :: block
    -> m (BlockFetchSendBlocks block m a)
    -> BlockFetchSendBlocks block m a

  -- | End of the stream of block bodies.
  --
  SendMsgBatchDone
    :: m (BlockFetchServer block m a)
    -> BlockFetchSendBlocks block m a
