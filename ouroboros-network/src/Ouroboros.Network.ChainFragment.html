--
-- * @l1@: the prefix of the first  chain fragment

-- * @r1@: The suffix of the first  chain fragment
-- * @r2@: The suffix of the second chain fragment
--
-- @l1@ and @l2@ will have same last block, i.e. /head/. The original chain

-- @

--
